router.post('/test-update', async (req, res) => {
    const { userId, newSolde } = req.body;
    
    // console.log(`[TEST UPDATE] üß™ Test de mise √† jour directe`);
    // console.log(`[TEST UPDATE] UserId: ${userId}, Nouveau solde: ${newSolde}`);
    
    if (!userId || newSolde === undefined) {
        return res.status(400).json({ message: "userId et newSolde requis" });
    }
    
    try {
        // 1. Lecture du solde actuel
        const [beforeRows] = await db.query("SELECT solde FROM user WHERE user_id = ?", [userId]);
        const soldeBefore = beforeRows.length > 0 ? beforeRows[0].solde : null;
        // console.log(`[TEST UPDATE] üìñ Solde avant mise √† jour: ${soldeBefore}`);
        
        // 2. Utilisation d'une connexion d√©di√©e avec transaction explicite
        const connection = await db.getConnection();
        try {
            await connection.beginTransaction();
            // console.log(`[TEST UPDATE] üîÑ Transaction d√©marr√©e`);
            
            // 3. UPDATE
            const [updateResult] = await connection.query(
                "UPDATE user SET solde = ? WHERE user_id = ?",
                [newSolde, userId]
            );
            
            // console.log(`[TEST UPDATE] üìä UPDATE Result:`, updateResult);
            // console.log(`[TEST UPDATE] üìä affectedRows: ${updateResult.affectedRows}, changedRows: ${updateResult.changedRows}`);
            
            // 4. COMMIT explicite
            await connection.commit();
            // console.log(`[TEST UPDATE] ‚úÖ Transaction commit√©e`);
            
            // 5. V√©rification avec m√™me connexion
            const [sameConnRows] = await connection.query("SELECT solde FROM user WHERE user_id = ?", [userId]);
            const soldeSameConn = sameConnRows.length > 0 ? sameConnRows[0].solde : null;
            // console.log(`[TEST UPDATE] üîç Solde avec m√™me connexion: ${soldeSameConn}`);
            
            // 6. Attendre un peu pour s'assurer de la persistence
            await new Promise(resolve => setTimeout(resolve, 100));
            
            // 7. V√©rification avec nouvelle connexion apr√®s d√©lai
            const [newConnRows] = await connection.query("SELECT solde FROM user WHERE user_id = ?", [userId]);
            const soldeNewConn = newConnRows.length > 0 ? newConnRows[0].solde : null;
            // console.log(`[TEST UPDATE] üîç Solde avec m√™me connexion apr√®s d√©lai: ${soldeNewConn}`);
            
        } finally {
            connection.release();
        }
        
        // 8. V√©rification finale avec pool
        const [finalRows] = await db.query("SELECT solde FROM user WHERE user_id = ?", [userId]);
        const soldeFinal = finalRows.length > 0 ? finalRows[0].solde : null;
        // console.log(`[TEST UPDATE] üîç Solde final avec pool: ${soldeFinal}`);
        
        res.json({
            success: true,
            soldeBefore,
            soldeExpected: newSolde,
            soldeFinal,
            status: soldeFinal === newSolde ? 'OK' : 'ERREUR'
        });
        
    } catch (error) {
        // console.error(`[TEST UPDATE] ‚ùå Erreur:`, error);
        res.status(500).json({ message: "Erreur lors du test" });
    }
});
 
/**
 * Fonction pour mettre √† jour les statistiques de roulette
 * Cette fonction est appel√©e apr√®s chaque partie pour enregistrer les r√©sultats
 */
async function updateStats(userId, payout) {
    // console.log(`[ROULETTE STATS] üéØ D√©but mise √† jour stats roulette (fonction directe)`);
    // console.log(`[ROULETTE STATS] UserId: ${userId}, Payout: ${payout}`);
    
    try {
        // D√©termine si c'est une victoire (payout positif)
        const isWin = payout > 0;
        // console.log(`[ROULETTE STATS] üé≤ R√©sultat: ${isWin ? 'VICTOIRE' : 'D√âFAITE'} (payout: ${payout})`);
        
        // V√©rifie s'il existe d√©j√† des statistiques pour cet utilisateur aujourd'hui
        // console.log(`[ROULETTE STATS] üîç Recherche stats existantes pour aujourd'hui...`);
        const [existingStats] = await db.query(
            `SELECT * FROM stats WHERE user_id = ? AND timestamp LIKE CONCAT(CURDATE(), '%')`,
            [userId]
        );
        
        // console.log(`[ROULETTE STATS] üìä Stats trouv√©es: ${existingStats.length} entr√©es`);
        
        if (existingStats.length > 0) {
            // Met √† jour les statistiques existantes pour aujourd'hui
            const currentStats = existingStats[0];
            const newNumGames = currentStats.num_games + 1;
            const newNumWins = currentStats.num_wins + (isWin ? 1 : 0);
            
            // console.log(`[ROULETTE STATS] üîÑ UPDATE - Anciens: ${currentStats.num_games} parties, ${currentStats.num_wins} victoires`);
            // console.log(`[ROULETTE STATS] üîÑ UPDATE - Nouveaux: ${newNumGames} parties, ${newNumWins} victoires`);
            
            await db.query(
                `UPDATE stats SET num_games = ?, num_wins = ?, timestamp = NOW() 
                 WHERE stat_id = ?`,
                [newNumGames, newNumWins, currentStats.stat_id]
            );
            
            // console.log(`[ROULETTE STATS] ‚úÖ UPDATE r√©ussi pour stat_id: ${currentStats.stat_id}`);
        } else {
            // Cr√©e une nouvelle entr√©e de statistiques
            // console.log(`[ROULETTE STATS] üÜï INSERT - Premi√®re partie du jour`);
            // console.log(`[ROULETTE STATS] üÜï INSERT - 1 partie, ${isWin ? 1 : 0} victoire`);
            
            await db.query(
                `INSERT INTO stats (user_id, num_games, num_wins, timestamp) 
                 VALUES (?, 1, ?, NOW())`,
                [userId, isWin ? 1 : 0]
            );
            
            // console.log(`[ROULETTE STATS] ‚úÖ INSERT r√©ussi pour user_id: ${userId}`);
        }
        
        // console.log(`[ROULETTE STATS] üéâ Stats mises √† jour avec succ√®s !`);
        
        return {
            success: true,
            game_played: 1,
            game_won: isWin ? 1 : 0,
            payout: payout
        };
        
    } catch (error) {
        console.error(`[ROULETTE STATS] ‚ùå Erreur lors de la mise √† jour des statistiques:`, error);
        throw error;
    }
}
